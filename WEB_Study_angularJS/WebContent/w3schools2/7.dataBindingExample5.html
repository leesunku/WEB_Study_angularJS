AngularJS Controller

<div ng-app="myApp" ng-controller="myCtrl"> <h1 ng-click="changeName()">{{firstName}}</h1> </div> <script type="text/javascript"> var app = angular.module('myApp', []); app.controller('myCtrl', function($scope) { $scope.firstName = "John"; $scope.changeName = function(){ $scope.firstName = "Nelly"; } }); </script>
