Modules and Controllers in Files

<div ng-app="myApp" ng-controller="myCtrl"> {{ firstName + " " + lastName }} </div> <script src="modulesExample5/myApp.js"></script> var app = angular.module("myApp", []); [] 파라메터에는 사용할 모듀를 정의하는 것, [] 파라메터 없으면, 새로운 모듈을 만들수 없다. 하지만 기존것을 사용한다. <script src="modulesExample5/myCtrl.js"></script> app.controller("myCtrl", function($scope){ $scope.firstName = "John"; $scope.lastName = "Doe"; }); GNS(Global Namespace)가 복잡할 수 있다. 자바스크립트에서는 클로벌 함수를 기피한다. 다른 스크립트로 인해 쉽게 덮어씌어 지거나 파괴되기 떄문이다. 앙귤라 보듈은 그런 문제를 감소시키기 위해, 모듈의 모든 펑션을 직역을 지킨다.
{{ firstName + " " + lastName }}
